        id bigint generated by default as identity,
        compoundHashMap varchar(255) for bit data,
        name varchar(255),
        compounds varchar(255) for bit data,
        primary key (id)
    )

        name varchar(255),

    )

    create table Nutrii.Meal (
        id varchar(255) not null,

        lifestyle varchar(255),
